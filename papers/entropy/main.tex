% ─── ABSTRACT ──────────────────────────────────────────────────────────────────
\begin{abstract}
For discrete distributions $p, q$ on a finite alphabet $\mathcal{X}$, the cross-entropy $H(p,q) = -\sum_{x} p(x) \log q(x)$ is infinite precisely when the violating set $V_{p,q} = \{x : p(x) > 0,\ q(x) = 0\}$ is non-empty, a strictly weaker condition than disjointness of supports.
We classify this singularity by the combinatorial size $|V_{p,q}|$, the mass severity $S(p,q) = \sum_{x \in V_{p,q}} p(x)$, and the asymptotic growth rate.
Under uniform smoothing $q^{(\varepsilon)} = (1-\varepsilon)q + \varepsilon u$, where $u$ is uniform on $|\mathcal{X}| = n$,
\[
  H(p, q^{(\varepsilon)}) = S(p,q)\,(-\log\varepsilon) + S(p,q)\log n + C_{p,q} + O(\varepsilon),
\]
so the rate of divergence equals the violating mass.
The Kullback--Leibler divergence inherits the singularity directly and is undefined at disjoint support---the probabilistic analog of vector orthogonality on $\mathbb{S}^{d-1}$, where the Euclidean distance is finite but the information-theoretic distance is not.
We further establish a per-coordinate asymmetry: contributions of the form $-p(x) \log q(x)$ are unbounded at $q(x) = 0,\ p(x) > 0$ and vanish at $p(x) = 0,\ q(x) > 0$.
Section~\ref{sec:alternatives} surveys divergences that are finite at the boundary.
A closing discussion (Section~\ref{sec:discussion}) examines how these properties bear on the design and behaviour of contemporary losses, treating that material as motivation and consequence rather than as further theorems.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% ═══════════════════════════════════════════════════════════════════════════════

The cross-entropy $H(p,q) = -\sum_x p(x) \log q(x)$ between discrete distributions is the standard divergence used to train statistical models, with the well-known property that minimisation in $q$ is equivalent to maximum likelihood and admits a unique minimiser $q = p$.
On the interior of the probability simplex it is a smooth and well-behaved functional.
On the boundary it is not: if $q$ assigns zero mass to a point where $p$ has positive mass, $H(p,q)$ is infinite.

Standard treatments record this only as a dichotomy: $H(p,q)$ is either finite or infinite, well-defined or undefined.
The present paper studies the structure of the singularity beyond this binary.
We show that the condition for divergence is strictly weaker than disjoint support; that the rate of divergence under smoothing is a real-valued function of the violating mass; that the per-coordinate contributions are infinitely asymmetric; and that the Kullback--Leibler divergence inherits these properties and fails to extend continuously to the configuration of disjoint support, which is the probabilistic analog of vector orthogonality.

The paper is organised as follows.
Section~\ref{sec:prelim} fixes notation.
Section~\ref{sec:taxonomy} states the divergence condition and the growth-rate identity under uniform smoothing.
Section~\ref{sec:asymmetry} establishes the per-coordinate asymmetry.
Section~\ref{sec:kl} treats the Kullback--Leibler divergence and its singularity at disjoint support, including the parallel with vector orthogonality.
Section~\ref{sec:alternatives} surveys divergence measures that remain finite at the boundary.
A closing discussion (Section~\ref{sec:discussion}) examines how the structure bears on losses used in contrastive and multimodal learning; that material is offered as motivation and consequence rather than as further theorems.

\section{Preliminaries}
\label{sec:prelim}
% ═══════════════════════════════════════════════════════════════════════════════

\begin{definition}[Cross-entropy]
\label{def:cross-entropy}
For discrete distributions $p$ and $q$ over a finite alphabet $\mathcal{X}$:
\begin{equation}
  H(p, q) = -\sum_{x \in \mathcal{X}} p(x) \log q(x),
  \label{eq:cross-entropy}
\end{equation}
with the convention $0 \log 0 = 0$ and $p(x) \log 0 = +\infty$ when $p(x) > 0$.
\end{definition}

\begin{definition}[KL divergence]
\label{def:kl}
The Kullback--Leibler divergence from $q$ to $p$ is:
\begin{equation}
  D_\mathrm{KL}(p \| q) = \sum_{x \in \mathcal{X}} p(x) \log \frac{p(x)}{q(x)} = H(p, q) - H(p),
  \label{eq:kl}
\end{equation}
where $H(p) = -\sum_x p(x) \log p(x)$ is the Shannon entropy.
\end{definition}

\begin{definition}[Support and violating set]
\label{def:support}
The \emph{support} of a distribution $p$ is $\mathrm{supp}(p) = \{x : p(x) > 0\}$.
The \emph{violating set} of $p$ with respect to $q$ is:
\begin{equation}
  V_{p,q} = \{x \in \mathcal{X} : p(x) > 0 \text{ and } q(x) = 0\} = \mathrm{supp}(p) \setminus \mathrm{supp}(q).
  \label{eq:violating}
\end{equation}
The \emph{violating mass} is $S(p,q) = \sum_{x \in V_{p,q}} p(x)$.
\end{definition}

\section{The Taxonomy of Singularities}
\label{sec:taxonomy}
% ═══════════════════════════════════════════════════════════════════════════════

\begin{theorem}[Necessary and sufficient condition for divergence]
\label{thm:divergence}
$H(p, q) = +\infty$ if and only if $V_{p,q} \neq \varnothing$, i.e., $\mathrm{supp}(p) \not\subseteq \mathrm{supp}(q)$.
\end{theorem}

\begin{proof}
($\Rightarrow$) If $V_{p,q} \neq \varnothing$, there exists $x_0$ with $p(x_0) > 0$ and $q(x_0) = 0$.
Then $-p(x_0) \log q(x_0) = +\infty$, so $H(p,q) = +\infty$.

($\Leftarrow$) If $V_{p,q} = \varnothing$, then $\mathrm{supp}(p) \subseteq \mathrm{supp}(q)$, so $q(x) > 0$ whenever $p(x) > 0$.
Each term $-p(x) \log q(x)$ is finite, and the sum over finite $\mathcal{X}$ is finite.
\end{proof}

\begin{critique}
The condition $V_{p,q} \neq \varnothing$ is strictly weaker than $\mathrm{supp}(p) \cap \mathrm{supp}(q) = \varnothing$.
Two distributions can share $99\%$ of their mass and still give $H(p,q) = +\infty$ provided a single coordinate exists where $p$ has positive mass and $q$ has none.
The textbook gloss ``cross-entropy diverges at disjoint support'' is a sufficient condition presented as if it were necessary and sufficient---and the distinction matters in practice, because models routinely produce $q$ with broadly overlapping but not subsuming support.
\end{critique}

\subsection{Example: Overlapping but Infinite}

\begin{casestudy}
Consider distributions over $\{A, B, C\}$:
\[
  p = (0.5,\; 0.3,\; 0.2), \qquad q = (0.7,\; 0.3,\; 0).
\]
The supports overlap on $\{A, B\}$, sharing $80\%$ of $p$'s mass.
But $V_{p,q} = \{C\}$ with $p(C) = 0.2 > 0$ and $q(C) = 0$:
\[
  H(p,q) = -0.5\log 0.7 - 0.3\log 0.3 - 0.2 \log 0 = +\infty.
\]
The supports are far from disjoint.
The infinity comes from a single uncovered coordinate.
\end{casestudy}

\subsection{The Three Tiers}

We propose a tiered classification of the singularity, going beyond the binary finite/infinite distinction.

\begin{definition}[Singularity tiers]
\label{def:tiers}
For distributions $p, q$ with $V_{p,q} \neq \varnothing$, define:
\begin{enumerate}[label=(\roman*)]
  \item \textbf{Combinatorial tier}: $|V_{p,q}|$ --- the number of violating coordinates.
  \item \textbf{Mass tier}: $S(p,q) = \sum_{x \in V_{p,q}} p(x)$ --- the total target mass on violating coordinates.
  \item \textbf{Rate tier}: the asymptotic growth rate of $H(p, q^{(\varepsilon)})$ as $\varepsilon \to 0^+$, where $q^{(\varepsilon)}$ is a smoothed approximation.
\end{enumerate}
\end{definition}

\begin{theorem}[Growth rate under uniform smoothing]
\label{thm:growth}
Let $q^{(\varepsilon)} = (1-\varepsilon)q + \varepsilon u$, where $u$ is the uniform distribution over $\mathcal{X}$ with $|\mathcal{X}| = n$.
Then as $\varepsilon \to 0^+$:
\begin{equation}
  \boxed{\;H(p, q^{(\varepsilon)}) = S(p,q)(-\log \varepsilon) + S(p,q)\log n + C_{p,q} + O(\varepsilon)\;}
  \label{eq:growth}
\end{equation}
where $C_{p,q} = -\sum_{x \notin V_{p,q}} p(x) \log q(x)$ is the finite part of the cross-entropy.
\end{theorem}

\begin{proof}
For $x \in V_{p,q}$: $q^{(\varepsilon)}(x) = \varepsilon/n$, so $-p(x)\log q^{(\varepsilon)}(x) = p(x)(-\log\varepsilon + \log n)$.
Summing over $V_{p,q}$: $S(p,q)(-\log\varepsilon + \log n)$.

For $x \notin V_{p,q}$: $q^{(\varepsilon)}(x) = (1-\varepsilon)q(x) + \varepsilon/n \to q(x)$ as $\varepsilon \to 0$, and $-p(x)\log q^{(\varepsilon)}(x) \to -p(x)\log q(x) = C_{p,q}^{(x)}$ with $O(\varepsilon)$ error.
\end{proof}

\begin{keyresult}
The divergence rate is exactly $S(p,q)$, the violating mass.
Not all infinities are equal:
\begin{center}
\begin{tabular}{@{}lccc@{}}
\toprule
Scenario & $|V|$ & $S(p,q)$ & Growth as $\varepsilon \to 0$ \\
\midrule
Single tail miss & 1 & 0.01 & $0.01(-\log\varepsilon)$ \\
Single mode miss & 1 & 0.90 & $0.90(-\log\varepsilon)$ \\
Complete disjoint & $n$ & 1.00 & $1.00(-\log\varepsilon)$ \\
\bottomrule
\end{tabular}
\end{center}
A model that misses a tail with $1\%$ mass diverges 100$\times$ slower than a model with completely disjoint support.
The violating mass $S(p,q)$ is the precise measure of singularity severity.
\end{keyresult}

\section{The Asymmetry of Cross-Entropy}
\label{sec:asymmetry}
% ═══════════════════════════════════════════════════════════════════════════════

Cross-entropy is not symmetric in its arguments: $H(p,q) \neq H(q,p)$ in general.
The asymmetry is not a notational curiosity.
It produces a sharp operational distinction at the boundary of the simplex, and it is the mechanism by which models trained under cross-entropy learn to prefer overcoverage to silence.

\begin{proposition}[Directional asymmetry]
\label{prop:asymmetry}
For distributions $p$ and $q$:
\begin{enumerate}[label=(\alph*)]
  \item If $q(x) = 0$ and $p(x) > 0$: \quad $-p(x)\log q(x) = +\infty$ \quad (infinite penalty).
  \item If $p(x) = 0$ and $q(x) > 0$: \quad $-p(x)\log q(x) = 0$ \quad (zero penalty).
\end{enumerate}
\end{proposition}

\begin{proof}
(a) follows from $\log 0 = -\infty$ and $p(x) > 0$.
(b) follows from $0 \cdot \log q(x) = 0$ by convention.
\end{proof}

\begin{geometricinsight}
This asymmetry has a clear operational interpretation:
\begin{itemize}
  \item \textbf{Missing truth ($q=0$ where $p>0$)}: The model is overconfident---it assigns zero probability to something that can happen.
  This is an infinite penalty.
  One missing coordinate is enough.
  \item \textbf{Adding falsehood ($q>0$ where $p=0$)}: The model is underconfident---it assigns probability to something that cannot happen.
  This is \emph{zero} penalty.
  The model can hallucinate with impunity.
\end{itemize}
Cross-entropy enforces \emph{coverage} with infinite force but imposes \emph{no constraint} on precision.
\end{geometricinsight}

\section{KL Divergence and the Impossible Zero}
\label{sec:kl}
% ═══════════════════════════════════════════════════════════════════════════════

Because $D_\mathrm{KL}(p\|q) = H(p,q) - H(p)$ and $H(p)$ is finite for any distribution on a finite alphabet, the singularity structure of KL divergence inherits directly from that of cross-entropy.

\begin{theorem}[KL divergence is undefined at disjoint support]
\label{thm:kl-undefined}
If $\mathrm{supp}(p) \cap \mathrm{supp}(q) = \varnothing$, then $D_\mathrm{KL}(p \| q) = +\infty$.
More generally, $D_\mathrm{KL}(p\|q) = +\infty$ if and only if $V_{p,q} \neq \varnothing$.
\end{theorem}

\begin{proposition}[Gradient explosion near the boundary]
\label{prop:gradient}
The gradient of $D_\mathrm{KL}(p\|q)$ with respect to $q_i$ (under the simplex constraint) is:
\begin{equation}
  \frac{\partial D_\mathrm{KL}}{\partial q_i} = -\frac{p_i}{q_i}.
  \label{eq:kl-gradient}
\end{equation}
As $q_i \to 0^+$ with $p_i > 0$, this gradient diverges: $|\nabla_{q_i}| \to +\infty$.
\end{proposition}

\begin{proof}
$D_\mathrm{KL}(p\|q) = \sum_x p(x)\log p(x) - \sum_x p(x)\log q(x)$.
Differentiating with respect to $q_i$: $\partial D_\mathrm{KL}/\partial q_i = -p_i/q_i$.
\end{proof}

\subsection{The Geometric Interpretation}

Disjoint support is the probabilistic analog of vector orthogonality.
Two distributions whose supports don't intersect are like two vectors with zero dot product: they share no common events, they occupy independent regions of the sample space.

\begin{keyresult}
\begin{center}
\begin{tabular}{@{}lcc@{}}
\toprule
& \textbf{Vector space} & \textbf{Probability space} \\
\midrule
Same & $\cos\theta = +1$ & $p = q$ \\
Independent & $\cos\theta = 0$ (orthogonal) & $\mathrm{supp}(p) \cap \mathrm{supp}(q) = \varnothing$ \\
\midrule
Distance at ``same'' & $d = 0$ (well-defined) & $D_\mathrm{KL} = 0$ (well-defined) \\
Distance at ``independent'' & $d = \sqrt{2}$ (well-defined) & $D_\mathrm{KL} = +\infty$ (\textbf{undefined}) \\
\bottomrule
\end{tabular}
\end{center}
Geometric orthogonality is a perfectly reachable configuration with a well-defined distance.
Probabilistic orthogonality is a singularity: the distance metric breaks at the very point that represents genuine independence.
\end{keyresult}

\section{Divergences Finite at the Boundary}
\label{sec:alternatives}
% ═══════════════════════════════════════════════════════════════════════════════

The singularity at disjoint support is not an inevitable feature of all divergence measures.
Several alternatives are well-defined at the boundary; we record their behaviour for reference.

\begin{table}[ht]
\centering
\caption{Divergence measures and their behaviour at disjoint support.}
\label{tab:alternatives}
\begin{tabular}{@{}lccc@{}}
\toprule
\textbf{Divergence} & \textbf{At disjoint supp.} & \textbf{Symmetric} & \textbf{Diff'able} \\
\midrule
Cross-entropy $H(p,q)$ & $+\infty$ & No & Yes (interior) \\
$D_\mathrm{KL}(p\|q)$ & $+\infty$ & No & Yes (interior) \\
Jensen--Shannon $\mathrm{JSD}$ & $\leq \log 2$ & Yes & Yes \\
Wasserstein $W_1(p,q)$ & Finite & Yes & Subgradient \\
MMD $\mathrm{MMD}^2(p,q)$ & Finite & Yes & Yes \\
Total variation $\mathrm{TV}(p,q)$ & $\leq 1$ & Yes & Subgradient \\
\bottomrule
\end{tabular}
\end{table}

Jensen--Shannon divergence is bounded by $\log 2$ and is well-defined everywhere; it was used in the original GAN formulation~\cite{goodfellow2014gan}.
Wasserstein distance~\cite{villani2009transport} does not require overlapping support and metrises weak convergence, making it the natural candidate for losses that need to measure distances at the boundary.
Maximum Mean Discrepancy operates in a reproducing kernel Hilbert space and is finite for any pair of distributions.
The cost paid for finiteness at the boundary differs across these alternatives---computational, statistical, and in terms of the gradient signal available to first-order optimisation---and a thorough comparison is beyond the scope of the present paper.

\section{Discussion}
\label{sec:discussion}
% ═══════════════════════════════════════════════════════════════════════════════

The preceding sections are mathematical content; this one is interpretive.
Three of the most persistent failure modes of modern multimodal and contrastive systems---the modality gap, hallucination, and InfoNCE's batch-size hunger---admit a natural reading in terms of that structure.
We present each as a \emph{hypothesis} unifying observed behaviour with the singularity, not as a derivation that displaces the existing accounts in the literature.
Each of these phenomena has competing explanations that any complete analysis must engage with; the singularity-shadow lens is one factor among them, offered here for its unifying value rather than as sole cause.

\subsection{The Modality Gap}

CLIP~\cite{radford2021clip} and SigLIP~\cite{zhai2023siglip} train image and text encoders to produce embeddings in a shared space.
Empirically, image and text embeddings do not overlap---they form two disjoint clusters separated by a persistent \emph{modality gap}~\cite{liang2022modality}.

\begin{remark}[The modality gap]
\label{rem:modality-gap}
Let $p_\text{img}$ and $p_\text{txt}$ denote the distributions of image and text embeddings on $\mathbb{S}^{d-1}$.
The contrastive loss wants mismatched pairs to have orthogonal embeddings, which in the distributional sense corresponds to pushing toward $\mathrm{supp}(p_\text{img}) \cap \mathrm{supp}(p_\text{txt}) = \varnothing$.
The cross-entropy-based loss diverges at this configuration (Theorem~\ref{thm:divergence}).
The optimisation therefore cannot reach orthogonal separation as a limit point; the residual ``safe distance'' it settles for is consistent with the empirically observed modality gap.
\end{remark}

\begin{geometricinsight}
The reading above is offered as a unifying hypothesis, not a sole-cause explanation.
The modality gap has at least three other live accounts in the literature.
\textbf{(i)} Liang et al.~\cite{liang2022modality} document a gap present at \emph{initialisation}, before any contrastive training, attributable to the geometry of randomly initialised deep encoders.
\textbf{(ii)} Cone effects in deep networks concentrate embeddings in narrow regions of the sphere irrespective of the contrastive objective.
\textbf{(iii)} Optimisation dynamics around the temperature parameter shape how aggressively the loss drives negatives across the equator.
The singularity-shadow account is consistent with each of these and may compose with them; a clean separation is, at this stage, an open empirical question.
\end{geometricinsight}

\subsection{Hallucination Under the Asymmetric Loss}

Large multimodal models hallucinate: they describe objects that do not exist in the image, invent facts, and produce confidently wrong outputs.

\begin{remark}[Overcoverage bias]
\label{rem:hallucination}
Let $p$ be the true distribution over outputs and $q$ the model's distribution.
At the coordinate level, cross-entropy contributions decompose into:
\begin{enumerate}[label=(\alph*)]
  \item if $q(x) = 0$ and $p(x) > 0$, the penalty is $+\infty$;
  \item if $q(x) > 0$ and $p(x) = 0$, the penalty is $0$.
\end{enumerate}
Per coordinate, putting mass on a real outcome that the model had zeroed eliminates the only unbounded term in the loss, while putting mass on an impossible outcome is, term by term, free.
This produces a systematic bias toward overcoverage under uncertainty.
\end{remark}

\begin{remark}[The simplex constraint qualifies the ``optimal strategy'' framing]
The per-coordinate result above is exact, but it does not on its own characterise an unconstrained optimum.
Under the softmax that produces $q$, the coordinates are coupled by the simplex constraint $\sum_x q(x) = 1$.
Mass placed on a falsehood necessarily comes from somewhere, including, in general, from real outcomes; ``adding falsehood'' is not literally free once one accounts for the displaced mass on truth.
The proposition therefore characterises a \emph{bias} rather than an \emph{optimum}: the direction in which gradients push under coverage-vs-precision tradeoffs, not the loss-minimising configuration of the model.
Stronger versions of the claim---such as ``hallucination is the optimal strategy under the loss''---require either dropping the simplex constraint or coupling the loss with regularisation that breaks the symmetry.
\end{remark}

\begin{casestudy}
The bias is recognisable in practice.
A model that distributes its output mass across 10 described objects in an image of 5 real objects pays a smaller marginal cross-entropy for the 5 hallucinated descriptions than the cost of zeroing out any of the 5 real ones.
The asymmetry is one mechanism among the empirical reasons multimodal models tend to overgenerate rather than undergenerate; it is consistent with the observed pattern but does not exhaust it.
\end{casestudy}

\subsection{The Computational Cost of Fighting the Singularity}

InfoNCE~\cite{oord2018infonce}---the loss behind CLIP and SimCLR~\cite{chen2020simclr}---computes a softmax cross-entropy over the batch similarity matrix.
For negative pairs, the loss pushes similarities toward the boundary where the singularity lives.

\begin{remark}[Singularity flatness and batch-size requirements]
\label{rem:batch-cost}
The InfoNCE gradient for negative pair $k$ is weighted by:
\begin{equation}
  w_k = \frac{\exp(\mathrm{sim}_k / \tau)}{\sum_j \exp(\mathrm{sim}_j / \tau)}.
\end{equation}
As negative similarities approach the singularity boundary (near orthogonality, $\mathrm{sim} \approx 0$), the softmax weights become approximately uniform, $w_k \approx 1/N$, and each negative contributes $O(1/N)$ gradient.
Accumulating useful signal in this regime requires $N$ large.
This is consistent with the large batches used in practice---for example, CLIP's $N = 32{,}768$---though it is one contributing factor among several.
\end{remark}

\begin{remark}
Other factors known to drive contrastive batch-size requirements compose with the singularity-flatness effect described above.
Hard-negative mining benefits from larger batches because the chance of seeing genuinely informative negatives grows with $N$; gradient variance reduction benefits from larger batches because the InfoNCE estimator's variance scales with $1/N$; the temperature schedule, the embedding dimension, and the choice of similarity function all shape the regime in which the softmax operates.
The proposition above identifies one mechanism within this composition; it does not claim to be the dominant one.
\end{remark}

\begin{keyresult}
The three phenomena---modality gap, hallucination bias, contrastive batch-size hunger---admit a unifying reading:
\begin{center}
\emph{Cross-entropy diverges at the configuration its consumers are trying to achieve.}
\end{center}
The modality gap is consistent with the optimisation maintaining a safe distance from the singularity, the asymmetry of cross-entropy biases models toward overcoverage under uncertainty, and the singularity-flatness effect is one driver of the gradient-signal pressure that demands large batches.
These are hypothesised structural causes; each phenomenon also has independent contributing accounts in the literature, with which the singularity reading composes rather than competes.
\end{keyresult}
